python mysql数据库连接

时间: 2016-03-21 12:41:24     来源: 财务狗

#!/usr/bin/python
# -*- coding: utf-8 -*-


import MySQLdb


# 打开数据库连接
db = MySQLdb.connect("localhost","testuser","test123","TESTDB" )


# 使用cursor()方法获取操作游标
cursor = db.cursor()


# SQL 查询语句
sql = "SELECT * FROM EMPLOYEE \
       WHERE INCOME > '%d'" % (1000)
try:
   # 执行SQL语句
   cursor.execute(sql)
   # 获取所有记录列表
   results = cursor.fetchall()
   for row in results:
      fname = row[0]
      lname = row[1]
      age = row[2]
      sex = row[3]
      income = row[4]
      # 打印结果
      print "fname=%s,lname=%s,age=%d,sex=%s,income=%d" % \
             (fname, lname, age, sex, income )
except:
   print "Error: unable to fecth data"
